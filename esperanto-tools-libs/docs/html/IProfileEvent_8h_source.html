173 template <class Archive> void load(Archive& ar, ProfileEvent& evt) {
174  std::string type;
175  ar(cereal::make_nvp("type", type));
176  evt.setType(type_from_string(type));
177 
178  std::string cls;
179  ar(cereal::make_nvp("class", cls));
180  evt.setClass(class_from_string(cls));
181 
182  ProfileEvent::TimePoint timeStamp;
183  ar(cereal::make_nvp("timeStamp", timeStamp));
184  evt.setTimeStamp(timeStamp);
185 
186  std::string threadId;
187  ar(cereal::make_nvp("thread_id", threadId));
188  auto tid = std::thread::id(std::stoull(threadId));
189  evt.setThreadId(tid);
190 
191  ProfileEvent::ExtraMetadata extra;
192  ar(cereal::make_nvp("extra", extra));
193  evt.setExtras(std::move(extra));
194 }
